def digits_operations(number):
    # Проверяем, что число двузначное
    if 10 <= number <= 99:
        # Получаем первую и вторую цифру числа
        first_digit = number // 10
        second_digit = number % 10

        # Вычисляем разность и частное
        difference = first_digit - second_digit
        # Проверяем вторую цифру, чтобы избежать деления на ноль
        if second_digit != 0:
            quotient = first_digit / second_digit
        else:
            quotient = None

        # Возвращаем результаты
        return difference, quotient
    else:
        return “Число не двузначное”, None

# Тестируем функцию
number = int(input(“Введите двузначное число: “))
difference, quotient = digits_operations(number)

if quotient is not None:
    print(f"Разность цифр числа: {difference}”)
    print(f"Частное цифр числа: {quotient}”)
else:
    print(difference)  # Выведет сообщение, что число не двузначное или деление на ноль```
